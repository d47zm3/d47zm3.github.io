Shrinking Go Binaries

# -s and -w flags

$ GOOS=linux go build cmd/go
$ ls -l go
-rwxr-xr-x  1 filippo  staff  12493536 Apr 16 16:58 go
$ GOOS=linux go build -ldflags="-s -w" cmd/go
$ ls -l go
-rwxr-xr-x  1 filippo  staff  8941952 Apr 16 17:08 go
