Git

Useful Commands

git config -l # list your config
git config --global user.name "Your Name"
git config --global user.email "user@example.com"

git init # start new repository

git log --oneline (--stat) # show commit log (and stats for changed files)
git log --patch-with-stat # show diffs on commits
git log --since="2019-10-10"
git log --since="2 days ago"
git log --until="2019-10-10"

git diff 60888d6941635a659197682efc369d47c1a09530 -- chart/values.yaml # show changes in file comparing to commit
git diff --staged # show changes in staged area

git commit -m "example commit message" # commit directly from command line
git branch (-a) # list branches (along with remote ones)
git fetch # download changes but do not merge them yet
git pull # git fetch + git merge

git merge develop # while on master branch, merge changes from develop branch

git rebase master # while on feature branch, rebase (re-apply) it's changes to master branch (add new nodes after current head of master)

git cherry-pick <commit id> # get commit id from feature branch (git log feature) and pick that commit and insert to current master branch

git commit --amend # fix last commit, change commit message or add forgotten file

git reset <file> # remove added file from staged area

git reset HEAD~ # reset to commit (HEAD~1 is default) removing commit and unstaging file (mixed mode, reset repo changes and stage area)
git reset --soft HEAD~ # remove commit but leave changes in current directory
git reset --hard HEAD~ # remove commit, do not leave any local changes

git reflog # show log of all git operations, merge, commits, checkout, pull...
git reset --hard HEAD@{3} # apply reset with identifier from reflog

git revert HEAD~ # reverts last commit and commit change

git rebase -i # interactive rebase, great tool to re-write commits

git stash # place current unstaged changes to stash
git stash ls # show all stashed list
git stash save refactor_func # save with comment
git stash show -p stash@{0} # show diff with stashed changes
git stash apply stash@{1}
git stash drop stash@{1}
git stash pop # apply last stash and remove it from stash
git stash clear # clear everything in stash
