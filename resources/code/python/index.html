# main
parser = argparse.ArgumentParser()
parser.add_argument("action", choices=['list', 'add', 'search', 'import'], help="list local repositories, add new one or search by tags")
parser.add_argument("--url", nargs='?', help="url of repository to add")
parser.add_argument("--note", nargs='?', help="personal note about repository")
parser.add_argument("--tags", nargs='?', help="tags for repository")
parser.add_argument("--clone", nargs='?', choices=['yes', 'no'], default="no", help="clone repository locally")
parser.add_argument("--user", nargs='?', help="user to download stars from")
args = parser.parse_args()
setup_yaml()

if args.action == "list":
    read_all()

elif args.action == "add":
    if args.url is None or args.note is None or args.tags is None:
        print("Error! Missing arguments for add!")
    else:
        add_repository(args.url, args.note, args.tags, args.clone)
elif args.action == "import":
    if args.user is None:
        print("Error! Missing arguments for import!")
    else:
        import_user_stars(args.user)
