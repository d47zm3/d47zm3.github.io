Create Github Pages Helm Repository


$ git clone git@github.com:d47zm3/helm-charts.git
$ cd helm-charts
$ mkdir src
$ helm create src/test-chart
$ helm package src/*
$ helm repo index --url https://d47zm3.me/helm-charts/ .
$ git add -A . && git commit -m "creating chart repository" && git push origin master
$ # set up github pages from master branch
$ helm repo add d47zm3 https://d47zm3.me/helm-charts/
$ # make changes
$ helm repo index --url https://d47zm3.me/helm-charts/ --merge index.yaml .
$ helm search repo test-chart
